163 def __init__(self, settings: Settings[EntityType], on_connect: Callable | None = None) -> None: 

164 """ 

165 Creates a basic discoverable object. 

166 

167 Args: 

168 settings: Settings for the entity we want to create in Home Assistant. 

169 See the `Settings` class for the available options. 

170 on_connect: Optional callback function invoked when the MQTT client \ 

171 successfully connects to the broker. 

172 If defined, you need to call `_connect_client()` to establish the \ 

173 connection manually. 

174 """ 

175 self._settings = settings 

176 self._entity = settings.entity 

177 

178 # Build the topic string: start from the type of component 

179 # e.g. `binary_sensor` 

180 self._entity_topic = f"{self._entity.component}" 

181 # If present, append the device name, e.g. `binary_sensor/mydevice` 

182 self._entity_topic += f"/{clean_string(self._entity.device.name)}" if self._entity.device else "" 

183 # Append the sensor name, e.g. `binary_sensor/mydevice/mysensor` 

184 self._entity_topic += f"/{clean_string(self._entity.name)}" 

185 

186 # Full topic where we publish the configuration message to be picked up by HA 

187 # Prepend the `discovery_prefix`, default: `homeassistant` 

188 # e.g. homeassistant/binary_sensor/mydevice/mysensor 

189 self.config_topic = f"{self._settings.mqtt.discovery_prefix}/{self._entity_topic}/config" 

190 # Full topic where we publish our own state messages 

191 # Prepend the `state_prefix`, default: `hmd` 

192 # e.g. hmd/binary_sensor/mydevice/mysensor 

193 self.state_topic = f"{self._settings.mqtt.state_prefix}/{self._entity_topic}/state" 

194 

195 # Full topic where we publish our own attributes as JSON messages 

196 # Prepend the `state_prefix`, default: `hmd` 

197 # e.g. hmd/binary_sensor/mydevice/mysensor 

198 self.attributes_topic = f"{self._settings.mqtt.state_prefix}/{self._entity_topic}/attributes" 

199 

200 logger.info(f"config_topic: {self.config_topic}") 

201 logger.info(f"state_topic: {self.state_topic}") 

202 if self._settings.manual_availability: 

203 # Define the availability topic, using `hmd` topic prefix 

204 self.availability_topic = f"{self._settings.mqtt.state_prefix}/{self._entity_topic}/availability" 

205 logger.debug(f"availability_topic: {self.availability_topic}") 

206 

207 # If the user has passed in a MQTT client, use it 

208 if self._settings.mqtt.client: 

209 self.mqtt_client = self._settings.mqtt.client 

210 else: 

211 # Create the MQTT client, registering the user `on_connect` callback 

212 self._setup_client(on_connect) 

213 # If there is a callback function defined, the user must manually connect 

214 # to the MQTT client 

215 if not on_connect: 

216 self._connect_client()

396class Subscriber(Discoverable[EntityType]): 

397 """ 

398 Specialized sub-lass that listens to commands coming from an MQTT topic 

399 """ 

400 

401 T = TypeVar("T") # Used in the callback function 

402 

403 def __init__( 

404 self, 

405 settings: Settings[EntityType], 

406 command_callback: Callable[[mqtt.Client, T, mqtt.MQTTMessage], Any], 

407 ) -> None: 

408 """ 

409 Entity that listens to commands from an MQTT topic. 

410 

411 Args: 

412 settings: Settings for the entity we want to create in Home Assistant. 

413 See the `Settings` class for the available options. 

414 command_callback: Callback function invoked when there is a command 

415 coming from the MQTT command topic 

416 """ 

417 

418 # Callback invoked when the MQTT connection is established 

419 def on_client_connected(client: mqtt.Client, *_): 

420 # Subscribe to the command topic 

421 result, _ = client.subscribe(self._command_topic, qos=1) 

422 if result is not mqtt.MQTT_ERR_SUCCESS: 422 ↛ 423line 422 didn't jump to line 423 because the condition on line 422 was never true

423 raise RuntimeError("Error subscribing to MQTT command topic") 

424 

425 # Invoke the parent init 

426 super().__init__(settings, on_client_connected) 

427 # Define the command topic to receive commands from HA, using `hmd` topic prefix 

428 self._command_topic = f"{self._settings.mqtt.state_prefix}/{self._entity_topic}/command" 

429 

430 # Register the user-supplied callback function 

431 self.mqtt_client.message_callback_add(self._command_topic, command_callback) 

432 

433 if self._settings.mqtt.client: 

434 # externally created MQTT client is used 

435 # which needs to be connected already 

436 # therefor explicitly subscribe to the command topic 

437 on_client_connected(self.mqtt_client) 

438 else: 

439 # Manually connect the MQTT client 

440 self._connect_client()
